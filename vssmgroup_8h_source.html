29 /***
30  * \brief groups events based on variable step size method
31  *
32  * This class groups the events based on the variable step size method.
33  * The waiting time is calculated from the total rate. The probability to
34  * choose an event is P_i = \omega_i / \omega_tot
35  *
36  * The template argument is a class which has to implement the functions
37  * Rate, which should return the rate of the event and
38  * onExecute, which is called if the event occurs.
39  * VSSMGroup implements these methods it's own and therefore
40  * can be used as an event itself. This allows hierarchical grouping
41  * of events.
42  *
43  * \todo add a refresh function if rates have changed
44  */
45 template<typename event_t>
46 class VSSMGroup {
47 public:
48  VSSMGroup() {
49  _acc_rate.push_back(0);
50  }
51 
52  /***
53  * \brief add event to list of possible events
54  */
55  void AddEvent(event_t *event) {
56  _events.push_back(event);
57  _acc_rate.push_back(_acc_rate.back() + event->Rate());
58  UpdateWaitingTime();
59  }
60 
64  double Rate() { return _acc_rate.back(); };
65 
69  void onExecute();
73  double WaitingTime() { return _waiting_time; }
74 
78  void UpdateWaitingTime() {
79  _waiting_time = -log( 1.0 - Random::rand_uniform() ) / Rate();
80  }
81 
82  event_t *getEvent(int i) { return _events[i]; }
83 
84 protected:
88  event_t *SelectEvent_LinearSearch();
89 
93  event_t *SelectEvent_BinarySearch();
94 
95  std::vector<event_t*> _events;
96  std::vector<double> _acc_rate;
97  double _waiting_time;
98 };

107 template<typename event_t>
108 inline event_t *VSSMGroup<event_t>::SelectEvent_LinearSearch()
109 {
110  double u = (1.-Random::rand_uniform())*Rate();
111  event_t *event;
112  // find the biggest event for that u < \sum omega_i
113  for(typename std::vector<event_t*>::iterator e=_events.begin();
114  e!=_events.end(); ++e) {
115  u-=(*e)->Rate();
116  if(u<=0) return *e;
117  }
118  // should never happen
119  return _events.back();
120 }

122 template<typename event_t>
123 event_t *VSSMGroup<event_t>::SelectEvent_BinarySearch()
124 {
125  double u = 1.-Random::rand_uniform();
126  u=u*Rate();
127  double max = Rate();
128  // to a binary search in accumulated events
129  int imin=0;
130  int imax=_acc_rate.size();
131  while(imax - imin > 1) {
132  int imid=(int)((imin+imax)*0.5);
133  //std::cout << u << " " << _acc_rate[imid] << std::endl;
134  if(u<=_acc_rate[imid])
135  imax=imid;
136  else
137  imin=imid;
138  }
139  //std::cout << imin << " " << Rate() << " " << u << std::endl;
140  return _events[imin];
141 }
