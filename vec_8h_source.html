196 inline std::istream &operator>>(std::istream &in, vec& v)
197 {
198  char c;
199  in.get(c);
200  if(c != '[') {
201  throw std::runtime_error("error, invalid character in vector string");
202  }
203 
204  string str;
205  while (in.good()) {
206  in.get(c);
207  if(c==']') { // found end of vector
208  Tokenizer tok(str, ",");
209  vector<double> d;
210  tok.ConvertToVector(d);
211  if(d.size() != 3)
212  throw std::runtime_error("error, invalid number of entries in vector");
213  v.setX(d[0]);
214  v.setY(d[1]);
215  v.setZ(d[2]);
216  return in;
217  }
218  str += c;
219  }
220  throw std::runtime_error("did not find closing bracket in string to vec conversion");
221 
222  return in;
223 }
