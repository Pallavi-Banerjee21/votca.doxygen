194  // serialization itself (template implementation stays in the header)
195  template<typename Archive>
196  void serialize(Archive& ar, const unsigned version) {
197 
198  // std::cout << "... ... Serializing the Orbitals." << std::endl ;
199  bool False = false;
200 
201  ar & _has_basis_set_size;
202  ar & _has_occupied_levels;
203  ar & _has_unoccupied_levels;
204  ar & _has_number_of_electrons;
205  ar & _has_level_degeneracy;
206  ar & _has_mo_energies;
207  ar & _has_atoms;
208  ar & _has_qm_energy;
209  ar & _has_self_energy;
210 
211  if ( _save_mo_coefficients ) { ar & _has_mo_coefficients; } else { ar & False; }
212  if ( _save_overlap ) { ar & _has_overlap; } else { ar & False; }
213 
214  if ( _has_basis_set_size ) { ar & _basis_set_size; }
215  if ( _has_occupied_levels ) { ar & _occupied_levels; }
216  if ( _has_unoccupied_levels ) { ar & _unoccupied_levels; }
217  if ( _has_number_of_electrons ) { ar & _number_of_electrons; }
218  if ( _has_level_degeneracy ) { ar & _level_degeneracy; }
219  if ( _has_mo_energies ) { ar & _mo_energies; }
220  if ( _has_mo_coefficients && _save_mo_coefficients ) { ar & _mo_coefficients; }
221  if ( _has_overlap && _save_overlap ) {
222  // symmetric matrix does not serialize by default
223  if (Archive::is_saving::value) {
224  unsigned size = _overlap.size1();
225  ar & size;
226  }
227 
228  // copy the values back if loading
229  if (Archive::is_loading::value) {
230  unsigned size;
231  ar & size;
232  _overlap.resize(size);
233  }
234 
235  for (unsigned i = 0; i < _overlap.size1(); ++i)
236  for (unsigned j = 0; j <= i; ++j)
237  ar & _overlap(i, j);
238  }
239 
240  if ( _has_atoms ) { ar & _atoms; }
241  if ( _has_qm_energy ) { ar & _qm_energy; }
242  if ( _has_self_energy ) { ar & _self_energy; }
243  }
244
