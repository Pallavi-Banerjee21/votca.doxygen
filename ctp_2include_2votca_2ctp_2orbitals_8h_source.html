261  // serialization itself (template implementation stays in the header)
262  template<typename Archive>
263  void serialize(Archive& ar, const unsigned version) {
264 
265  ar & _has_basis_set_size;
266  ar & _has_occupied_levels;
267  ar & _has_unoccupied_levels;
268  ar & _has_number_of_electrons;
269  ar & _has_level_degeneracy;
270  ar & _has_mo_energies;
271  ar & _has_mo_coefficients;
272  ar & _has_overlap;
273  ar & _has_atoms;
274  ar & _has_qm_energy;
275  ar & _has_self_energy;
276  ar & _has_integrals;
277 
278  //cout << "\nLoaded 1\n" << flush;
279  if ( _has_basis_set_size ) { ar & _basis_set_size; }
280  if ( _has_occupied_levels ) { ar & _occupied_levels; }
281  if ( _has_unoccupied_levels ) { ar & _unoccupied_levels; }
282  if ( _has_number_of_electrons ) { ar & _number_of_electrons; }
283  if ( _has_level_degeneracy ) { ar & _level_degeneracy; }
284  if ( _has_mo_energies ) { ar & _mo_energies; }
285  if ( _has_mo_coefficients ) { ar & _mo_coefficients; }
286  if ( _has_overlap ) {
287  // symmetric matrix does not serialize by default
288  if (Archive::is_saving::value) {
289  unsigned size = _overlap.size1();
290  ar & size;
291  }
292 
293  // copy the values back if loading
294  if (Archive::is_loading::value) {
295  unsigned size;
296  ar & size;
297  _overlap.resize(size);
298  }
299 
300  for (unsigned i = 0; i < _overlap.size1(); ++i)
301  for (unsigned j = 0; j <= i; ++j)
302  ar & _overlap(i, j);
303  }
304 
305  if ( _has_atoms ) { ar & _atoms; }
306  if ( _has_qm_energy ) { ar & _qm_energy; }
307  if ( _has_self_energy ) { ar & _self_energy; }
308  if ( _has_integrals ) { ar & _integrals; }
309 
310  //cout << "\nLoaded 2\n" << flush;
311  // GW-BSE storage
312  if(version > 0) {
313  ar & _has_vxc;
314  ar & _has_QPpert;
315  ar & _has_QPdiag;
316  ar & _has_BSE_singlets;
317  ar & _has_BSE_triplets;
318  if ( _has_QPpert ) { ar & _QP_levels_index; ar & _QPpert_energies; }
319  if ( _has_QPdiag ) { ar & _QPdiag_energies; ar & _QPdiag_coefficients; }
320  if ( _has_BSE_singlets || _has_BSE_triplets ) { ar & _BSE_levels_indices; }
321  if ( _has_BSE_singlets ) { ar & _BSE_singlet_energies; ar & _BSE_singlet_coefficients; }
322  if ( _has_BSE_triplets ) { ar & _BSE_triplet_energies; ar & _BSE_triplet_coefficients; }
323 
324  //cout << "\nLoaded 3\n" << flush;
325  if ( _has_vxc ) {
326  // symmetric matrix does not serialize by default
327  if (Archive::is_saving::value) {
328  unsigned size = _vxc.size1();
329  ar & size;
330  }
331 
332  // copy the values back if loading
333  if (Archive::is_loading::value) {
334  unsigned size;
335  ar & size;
336  _vxc.resize(size);
337  }
338 
339  for (unsigned i = 0; i < _vxc.size1(); ++i)
340  for (unsigned j = 0; j <= i; ++j)
341  ar & _vxc(i, j);
342  }
343  //cout << "\nLoaded 4\n" << flush;
344 
345  } // end version 1: GW-BSE storage
346  }// end of serialization
