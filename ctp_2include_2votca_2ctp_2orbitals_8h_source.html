252  // serialization itself (template implementation stays in the header)
253  template<typename Archive>
254  void serialize(Archive& ar, const unsigned version) {
255 
256  ar & _has_basis_set_size;
257  ar & _has_occupied_levels;
258  ar & _has_unoccupied_levels;
259  ar & _has_number_of_electrons;
260  ar & _has_level_degeneracy;
261  ar & _has_mo_energies;
262  ar & _has_mo_coefficients;
263  ar & _has_overlap;
264  ar & _has_atoms;
265  ar & _has_qm_energy;
266  ar & _has_self_energy;
267  ar & _has_integrals;
268 
269  //cout << "\nLoaded 1\n" << flush;
270  if ( _has_basis_set_size ) { ar & _basis_set_size; }
271  if ( _has_occupied_levels ) { ar & _occupied_levels; }
272  if ( _has_unoccupied_levels ) { ar & _unoccupied_levels; }
273  if ( _has_number_of_electrons ) { ar & _number_of_electrons; }
274  if ( _has_level_degeneracy ) { ar & _level_degeneracy; }
275  if ( _has_mo_energies ) { ar & _mo_energies; }
276  if ( _has_mo_coefficients ) { ar & _mo_coefficients; }
277  if ( _has_overlap ) {
278  // symmetric matrix does not serialize by default
279  if (Archive::is_saving::value) {
280  unsigned size = _overlap.size1();
281  ar & size;
282  }
283 
284  // copy the values back if loading
285  if (Archive::is_loading::value) {
286  unsigned size;
287  ar & size;
288  _overlap.resize(size);
289  }
290 
291  for (unsigned i = 0; i < _overlap.size1(); ++i)
292  for (unsigned j = 0; j <= i; ++j)
293  ar & _overlap(i, j);
294  }
295 
296  if ( _has_atoms ) { ar & _atoms; }
297  if ( _has_qm_energy ) { ar & _qm_energy; }
298  if ( _has_self_energy ) { ar & _self_energy; }
299  if ( _has_integrals ) { ar & _integrals; }
300 
301  //cout << "\nLoaded 2\n" << flush;
302  // GW-BSE storage
303  if(version > 0) {
304  ar & _has_vxc;
305  ar & _has_QPpert;
306  ar & _has_QPdiag;
307  ar & _has_BSE_singlets;
308  ar & _has_BSE_triplets;
309  if ( _has_QPpert ) { ar & _QP_levels_index; ar & _QPpert_energies; }
310  if ( _has_QPdiag ) { ar & _QPdiag_energies; ar & _QPdiag_coefficients; }
311  if ( _has_BSE_singlets || _has_BSE_triplets ) { ar & _BSE_levels_indices; }
312  if ( _has_BSE_singlets ) { ar & _BSE_singlet_energies; ar & _BSE_singlet_coefficients; }
313  if ( _has_BSE_triplets ) { ar & _BSE_triplet_energies; ar & _BSE_triplet_coefficients; }
314 
315  //cout << "\nLoaded 3\n" << flush;
316  if ( _has_vxc ) {
317  // symmetric matrix does not serialize by default
318  if (Archive::is_saving::value) {
319  unsigned size = _vxc.size1();
320  ar & size;
321  }
322 
323  // copy the values back if loading
324  if (Archive::is_loading::value) {
325  unsigned size;
326  ar & size;
327  _vxc.resize(size);
328  }
329 
330  for (unsigned i = 0; i < _vxc.size1(); ++i)
331  for (unsigned j = 0; j <= i; ++j)
332  ar & _vxc(i, j);
333  }
334  //cout << "\nLoaded 4\n" << flush;
335 
336  } // end version 1: GW-BSE storage
337  }// end of serialization
