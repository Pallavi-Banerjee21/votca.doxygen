26 bool ESPTopologyReader::ReadTopology(string file, Topology &top)
27 {
28  // cleanup topology to store new data
29  ifstream fl;
30  string parse_line, type, mass, tmp, virt;
31  unsigned int white_space1, white_space2;
32  int num_molecules;
33  unsigned int check_name, check_name2;
34  int *num_atoms = NULL;
35  double box_x,box_y,box_z;
36  bool HasMass, HasVirtual;
37  top.Cleanup();
38 
39  fl.open(file.c_str());
40  if(!fl.is_open())
41  return false;
42 
43  // 'box_l' variable for box geometry
45  getline(fl, parse_line);
46  parse_line = ReadBlockfileLine(parse_line, "box_l");
47  white_space1 = parse_line.find(' ');
48  if (white_space1 != string::npos) {
49  white_space2 = parse_line.find(' ',white_space1+1);
50  if (white_space2 != string::npos) {
51  box_x = atof(parse_line.substr(0,white_space1).c_str());
52  box_y = atof(parse_line.substr(white_space1+1,white_space2).c_str());
53  box_z = atof(parse_line.substr(white_space2).c_str());
54  } else {
55  cerr << "Error in parsing blockfile (box_l).\n";
56  return false;
57  }
58  } else {
59  cerr << "Error in parsing blockfile (box_l).\n";
60  return false;
61  }
62  top.setBox(matrix(vec(box_x, 0, 0), vec(0, box_y, 0), vec(0, 0, box_z)));
63  //cout << "Box geometry: " << box_x << " " << box_y << " " << box_z << endl;
64 
65  // 'num_molecules' tclvariable for total number of molecules
67  getline(fl, parse_line);
68  parse_line = ReadBlockfileLine(parse_line, "num_molecules");
69  num_molecules = atof(parse_line.c_str());
70  //cout << "Number of molecules: " << num_molecules << endl;
71 
72  // 'num_atoms' tclvariable for list of number of atoms (for each molecule)
74  num_atoms = (int*) calloc(num_molecules,sizeof(int));
75  getline(fl, parse_line);
76  parse_line = ReadBlockfileLine(parse_line, "num_atoms");
77  for (int i=0; i<num_molecules-1;++i) {
78  white_space1 = parse_line.find(' ');
79  if (white_space1 != string::npos) {
80  num_atoms[i] = atoi(parse_line.substr(0,white_space1+1).c_str());
81  parse_line = parse_line.substr(white_space1+1);
82  } else {
83  cerr << "Error in parsing blockfile (num_atoms).\n";
84  return false;
85  }
86  }
87  num_atoms[num_molecules-1] = atoi(parse_line.c_str());
88 
89  // 'particles' check whether mass is provided (if so, at the end)
91  getline(fl, parse_line);
92  // make sure we're at 'particles'
93  check_name = parse_line.find("{particles");
94  if (check_name == string::npos) {
95  cerr << "Can't find particles variable in blockfile.\n";
96  return false;
97  }
98  check_name = parse_line.find("mass");
99  check_name2 = parse_line.find("virtual");
100  if (check_name != string::npos) {
101  HasMass = 1;
102  if (check_name2 != string::npos) {
103  HasVirtual = 1;
104  // make sure the format is correct
105  check_name = parse_line.find("{id type molecule mass virtual pos v}");
106  check_name2 = parse_line.find("{id type molecule mass virtual folded_position v}");
107  } else {
108  HasVirtual = 0;
109  // make sure the format is correct
110  check_name = parse_line.find("{id type molecule mass pos v}");
111  check_name2 = parse_line.find("{id type molecule mass folded_position v}");
112  }
113  } else {
114  HasMass = 0;
115  if (check_name2 != string::npos) {
116  HasVirtual = 1;
117  // make sure the format is correct
118  check_name = parse_line.find("{id type molecule virtual pos v}");
119  check_name2 = parse_line.find("{id type molecule virtual folded_position v}");
120  } else {
121  HasVirtual = 0;
122  // make sure the format is correct
123  check_name = parse_line.find("{id type molecule pos v}");
124  check_name2 = parse_line.find("{id type molecule folded_position v}");
125  }
126  }
127  if (check_name == string::npos && check_name2 == string::npos) {
128  cerr << "Check format of particles variable in blockfile.\n"
129  "Should be {id type molecule [mass] pos v}.\n"
130  "Instead: " << parse_line << endl;
131  return false;
132  }
133 
134  // At this point we're assuming the particles have been ordered wrt molecules
135  for (int mol=0; mol<num_molecules; ++mol) {
136  Molecule *mi = top.CreateMolecule(boost::lexical_cast<string>(mol));
137  for (int atom=0; atom<num_atoms[mol]; ++atom) {
138  fl >> tmp;
139  fl >> type;
140  fl >> tmp;
141  if (HasMass)
142  fl >> mass;
143  else
144  mass = string("1.0");
145 
146 
147  mi->AddBead(top.CreateBead(1, "", top.GetOrCreateBeadType(type), 0, atoi(mass.c_str()), 0),
148  boost::lexical_cast<string>(atom));
149  getline(fl, parse_line);
150  }
151  }
152 
153  // Check that we've reached the end of the 'particles' variable
154  getline(fl, parse_line);
155  if (parse_line != "}") {
156  cerr << "'num_molecules' and 'num_atoms' do not correspond to "
157  "number of particles. Check .esp file.\n";
158  return false;
159  }
160 
161 
162  free(num_atoms);
163 
164  fl.close();
165 
166  cout << "WARNING: topology created from .esp file, charges aren't loaded!\n";
167 
168  return true;
169 }

172 string ESPTopologyReader::ReadBlockfileLine(string input_line, string variable)
173 {
174  unsigned int pos_openvar, pos_closevar;
175 
176  string str_find = "{" + variable;
177 
178  pos_openvar = input_line.find(str_find);
179  if (pos_openvar == string::npos) {
180  throw runtime_error("Can't find '" + variable + "' variable in blockfile.\n");
181  }
182  input_line = input_line.substr(pos_openvar+variable.length()+2);
183  pos_closevar = input_line.find('}');
184  if (pos_closevar == string::npos) {
185  throw runtime_error("Can't find '}' character in closing '" + variable + "' variable of the blockfile.\n");
186  }
187  input_line = input_line.substr(0,pos_closevar);
188 
189  return input_line;
190 }
