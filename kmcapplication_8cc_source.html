64 void KMCApplication::PrintDescription(const char *name, const bool length)
65 {
66  // loading the documentation xml file from VOTCASHARE
67  char *votca_share = getenv("VOTCASHARE");
68  if(votca_share == NULL) throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
69  string xmlFile = string(getenv("VOTCASHARE")) + string("/kmc/xml/")+name+string(".xml");
70  try {
71  Property options;
72  load_property_from_xml(options, xmlFile);
73 
74  if ( length ) { // short description of the calculator
75 
76  cout << string(" ") << _fwstring(string(name),14);
77  cout << options.get("options."+string(name)).getAttribute<string>("help");
78 
79  } else { // long description of the calculator
80  votca::tools::PropertyIOManipulator iom(votca::tools::PropertyIOManipulator::HLP, 2, "");
81  cout << iom << options;
82  }
83  cout << endl;
84  } catch(std::exception &error) {
85  cout << string("XML file or description tag missing: ") << xmlFile << endl;
86  }
87 }

101 
102 
103  if(OptionsMap().count("description")) {
104  CheckRequired("description", "no calculator is given");
105  Tokenizer tok(OptionsMap()["description"].as<string>(), " ,\n\t");
106  // loop over the names in the description string
107  for (Tokenizer::iterator n = tok.begin(); n != tok.end(); ++n) {
108  // loop over calculators
109  bool printerror = true;
110  for(KMCCalculatorFactory::assoc_map::const_iterator iter=Calculators().getObjects().begin();
111  iter != Calculators().getObjects().end(); ++iter) {
112 
113  if ( (*n).compare( (iter->first).c_str() ) == 0 ) {
114  PrintDescription( (iter->first).c_str(), _long );
115  printerror = false;
116  break;
117  }
118  }
119  if ( printerror ) cout << "Calculator " << *n << " does not exist\n";
120  }
121  StopExecution();
122  return true;
123  }

126 
127  CheckRequired("execute", "no calculator is given");
128  Tokenizer tok(OptionsMap()["execute"].as<string>(), " ,\n\t");

130  AddCalculator(Calculators().Create((*n).c_str()));

132  CheckRequired("options", "please provide an xml file with program options");
