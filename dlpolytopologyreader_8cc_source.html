32 bool DLPOLYTopologyReader::ReadTopology(string file, Topology &top)
33 {
34  std::ifstream fl;
35  boost::filesystem::path filepath(file.c_str());
36  string filename;
37 
38  // TODO: fix residue naming / assignment
39  Residue *res = top.CreateResidue("no");
40 
41  if ( boost::filesystem::basename(filepath).size() == 0 ) {
42  if (filepath.parent_path().string().size() == 0) {
43  filename="FIELD";
44  } else {
45  filename=filepath.parent_path().string() + "/FIELD";
46  }
47  } else {
48  filename=file;
49  }
50  fl.open(filename.c_str());
51  if (!fl.is_open()){
52  throw std::runtime_error("could open dlpoly file " + filename);
53  } else {
54  string line;
55  getline(fl, line); //title
56  getline(fl, line); //unit line
57  fl >> line;
58  if (line != "MOLECULES")
59  throw std::runtime_error("unexpected line in dlpoly file " + filename + ", expected 'MOLECULES' got" + line);
60  int nmol_types;
61  fl >> nmol_types;
62  getline(fl, line); //rest of MOLECULES line
63  int id=0;
64  for (int nmol_type=0;nmol_type<nmol_types; nmol_type++){
65  string mol_name;
66  getline(fl, mol_name); //molecule name might incl. spaces
67  boost::erase_all(mol_name, " ");
68  Molecule *mi = top.CreateMolecule(mol_name);
69  fl >> line;
70  if (line != "NUMMOLS")
71  throw std::runtime_error("unexpected line in dlpoly file " + filename + ", expected 'NUMMOLS' got" + line);
72  int nreplica;
73  fl >> nreplica;
74  fl >> line;
75  if (line != "ATOMS")
76  throw std::runtime_error("unexpected line in dlpoly file " + filename + ", expected 'ATOMS' got" + line);
77  int natoms;
78  fl >> natoms;
79  //read molecule
80  int id_map[natoms];
81  for (int i=0;i<natoms;){//i is altered in reapeater loop
82  string beadtype;
83  fl >> beadtype;
84  BeadType *type = top.GetOrCreateBeadType(beadtype);
85  double mass;
86  fl >> mass;
87  double charge;
88  fl >> charge;
89  getline(fl,line); //rest of the atom line
90  Tokenizer tok(line, " ");
91  vector<int> fields;
92  tok.ConvertToVector<int>(fields);
93  int repeater=1;
94  if (fields.size() > 1)
95  repeater=fields[0];
96  for(int j=0;j<repeater;j++){
97  string beadname = beadtype + "#" + boost::lexical_cast<string>(j+1);
98  Bead *bead = top.CreateBead(1, beadname , type, res->getId(), mass, charge);
99  stringstream nm;
100  nm << bead->getResnr() + 1 << ":" << top.getResidue(bead->getResnr())->getName() << ":" << bead->getName();
101  mi->AddBead(bead, nm.str());
102  id_map[i]=bead->getId();
103  i++;
104  }
105  }
106  while (line != "FINISH"){
107  if ((line == "BONDS")||(line == "ANGLES")||(line == "DIHEDRALS")) {
108  string type = line;
109  int count;
110  fl >> count;
111  for (int i=0;i<count;i++){
112  fl >> line; //bond/angle/dih type not used
113  int ids[4];
114  Interaction *ic;
115  fl >> ids[0]; fl>>ids[1];
116  if (type == "BONDS"){
117  ic = new IBond(id_map[ids[0]-1],id_map[ids[1]-1]); // -1 due to fortran vs c
118  } else if (type == "ANGLES"){
119  fl >> ids[2];
120  ic = new IAngle(id_map[ids[0]-1],id_map[ids[1]-1],id_map[ids[2]-1]); // -1 due to fortran vs c
121  } else if (type == "DIHEDRALS"){
122  fl >> ids[2]; fl >> ids[3];
123  ic = new IDihedral(id_map[ids[0]-1],id_map[ids[1]-1],id_map[ids[2]-1],id_map[ids[3]-1]); // -1 due to fortran vs c
124  }
125  ic->setGroup(type);
126  ic->setIndex(i);
127  ic->setMolecule(mi->getId());
128  top.AddBondedInteraction(ic);
129  mi->AddInteraction(ic);
130  getline(fl,line);
131  }
132  }
133  fl >> line;
134  if (fl.eof())
135  throw std::runtime_error("unexpected end of dlpoly file " + filename + " while scanning for 'FINISH'");
136  }
137  getline(fl, line); //rest of the FINISH line
138  //replicate molecule
139  for (int replica=1;replica<nreplica;replica++){
140  Molecule *mi_replica = top.CreateMolecule(mol_name);
141  for(int i=0;i<mi->BeadCount();i++){
142  Bead *bead=mi->getBead(i);
143  BeadType *type = top.GetOrCreateBeadType(bead->Type()->getName());
144  string beadname=mi->getBeadName(i);
145  Bead *bead_replica = top.CreateBead(1, bead->getName(), type, res->getId(), bead->getM(), bead->getQ());
146  mi_replica->AddBead(bead_replica,beadname);
147  }
148  InteractionContainer ics=mi->Interactions();
149  for(vector<Interaction *>::iterator ic=ics.begin(); ic!=ics.end(); ++ic) {
150  Interaction *ic_replica;
151  //TODO: change if beads are not continous anymore
152  int offset = mi_replica->getBead(0)->getId() - mi->getBead(0)->getId();
153  if ((*ic)->BeadCount() == 2) {
154  ic_replica = new IBond((*ic)->getBeadId(0)+offset,(*ic)->getBeadId(1)+offset);
155  } else if ((*ic)->BeadCount() == 3) {
156  ic_replica = new IAngle((*ic)->getBeadId(0)+offset,(*ic)->getBeadId(1)+offset,(*ic)->getBeadId(2)+offset);
157  } else if ((*ic)->BeadCount() == 4) {
158  ic_replica = new IDihedral((*ic)->getBeadId(0)+offset,(*ic)->getBeadId(1)+offset,(*ic)->getBeadId(2)+offset,(*ic)->getBeadId(3)+offset);
159  }
160  ic_replica->setGroup((*ic)->getGroup());
161  ic_replica->setIndex((*ic)->getIndex());
162  ic_replica->setMolecule(mi_replica->getId());
163  top.AddBondedInteraction(ic_replica);
164  mi_replica->AddInteraction(ic_replica);
165  }
166  }
167  }
168  top.RebuildExclusions();
169  }
170  //we don't need the rest.
171  fl.close();
172 
173  if ( boost::filesystem::basename(filepath).size() == 0 ) {
174  if (filepath.parent_path().string().size() == 0) {
175  filename="CONFIG";
176  } else {
177  filename=filepath.parent_path().string() + "/CONFIG";
178  }
179  } else {
180  cout << "NOTE: Explicit dlploy topology filename given, so no config will be openend and no boundary conditions will set in the topology." << endl;
181  return true;
182  }
183  fl.open(filename.c_str());
184  if(fl.is_open()) {
185  string line;
186  getline(fl, line); //title
187  getline(fl, line); // 2nd header line
188  vec box_vectors[3];
189  for (int i=0;i<3;i++){ // read 3 box lines
190  getline(fl, line);
191  if(fl.eof())
192  throw std::runtime_error("unexpected end of file in dlpoly file " + filename +", when reading box vector" +
193  boost::lexical_cast<string>(i));
194  Tokenizer tok(line, " ");
195  vector<double> fields;
196  tok.ConvertToVector<double>(fields);
197  box_vectors[i]=vec(fields[0],fields[1],fields[2]);
198  }
199  matrix box(box_vectors[0],box_vectors[1],box_vectors[2]);
200  top.setBox(box);
201  fl.close();
202  }
203  else {
204  cout << "NOTE: Could not open dlploy file " << filename << "so no boundary conditions, where set in the topology." << endl;
205  }
206  return true;
207 }
