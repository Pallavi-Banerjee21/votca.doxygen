309 double IDFT::getCouplingElement( int levelA, int levelB, Orbitals* _orbitalsA,
310  Orbitals* _orbitalsB, ub::matrix<double>* _JAB ) {
311 
312  int _levelsA = _orbitalsA->getNumberOfLevels();
313  int _levelsB = _orbitalsB->getNumberOfLevels();
314 
315  if ( _energy_difference != 0 ) {
316  std::vector<int> list_levelsA = *_orbitalsA->getDegeneracy( levelA, _energy_difference );
317  std::vector<int> list_levelsB = *_orbitalsA->getDegeneracy( levelB, _energy_difference );
318 
319  double _JAB_sq = 0; double _JAB_one_level;
320 
321  for (std::vector<int>::iterator iA = list_levelsA.begin()++; iA != list_levelsA.end(); iA++) {
322  for (std::vector<int>::iterator iB = list_levelsB.begin()++; iB != list_levelsB.end(); iB++) {
323  //cout << *iA << ':' << *iB << endl;
324  _JAB_one_level = _JAB->at_element( *iA - 1 , *iB -1 + _levelsA );
325  _JAB_sq += _JAB_one_level*_JAB_one_level ;
326  }
327  }
328 
329  return sqrt(_JAB_sq / ( list_levelsA.size() * list_levelsB.size() ) ) * _conv_Hrt_eV ;
330 
331  } else {
332  return _JAB->at_element( levelA - 1 , levelB -1 + _levelsA ) * _conv_Hrt_eV;
333  }
334  // the matrix should be symmetric, could also return this element
335  // _JAB.at_element( _levelsA + levelB - 1 , levelA - 1 );
336 }

476 void IDFT::PrepareGuess( Orbitals* _orbitalsA, Orbitals* _orbitalsB,

486  int _levelsB = _orbitalsB->getNumberOfLevels();

489  int _electronsB = _orbitalsB->getNumberOfElectrons();

491  ub::zero_matrix<double> zeroB( _levelsA, _basisB ) ;
492  ub::zero_matrix<double> zeroA( _levelsB, _basisA ) ;
493 
494  ub::matrix<double>* _mo_coefficients = _orbitalsAB->getOrbitals();
495  //cout << "MO coefficients " << *_mo_coefficients << endl;

501  _orbitalsAB->setNumberOfLevels( _electronsA - _electronsB ,
502  _levelsA + _levelsB - _electronsA - _electronsB );
503  _orbitalsAB->setNumberOfElectrons( _electronsA + _electronsB );
504 
505  ub::project( *_mo_coefficients, ub::range (0, _levelsA ), ub::range ( _basisA, _basisA +_basisB ) ) = zeroB;
506  ub::project( *_mo_coefficients, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( 0, _basisA ) ) = zeroA;
507  ub::project( *_mo_coefficients, ub::range (0, _levelsA ), ub::range ( 0, _basisA ) ) = *_orbitalsA->getOrbitals();
508  ub::project( *_mo_coefficients, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( _basisA, _basisA + _basisB ) ) = *_orbitalsB->getOrbitals();
