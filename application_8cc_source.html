87  }

95 boost::program_options::options_description_easy_init
96  Application::AddProgramOptions(const string &group)
97 {
98  // if no group is given, add it to standard options
99  if(group == "")
100  return _op_desc.add_options();
101 
102  // does group already exist, if yes, add it there
103  std::map<string, boost::program_options::options_description>::iterator iter;
104  iter = _op_groups.find(group);
105  if(iter!=_op_groups.end())
106  return iter->second.add_options();
107 
108  // no group with given name was found -> create group
109  _op_groups.insert(make_pair(group, boost::program_options::options_description(group)));
110 
111  return _op_groups[group].add_options();
112 }

115 void Application::ParseCommandLine(int argc, char **argv)
116 {
117  namespace po = boost::program_options;
118 
119  std::map<string, boost::program_options::options_description>::iterator iter;
120 
121  // add all cathegories to list of available options
122  for(iter=_op_groups.begin(); iter!=_op_groups.end(); ++iter)
123  _op_desc.add(iter->second);
124 
125  // parse the command line
126  try {
127  po::store(po::parse_command_line(argc, argv, _op_desc), _op_vm);
128  po::notify(_op_vm);
129  }
130  catch(boost::program_options::error err) {
131  throw runtime_error(string("error parsing command line: ") + err.what());
132  }
133 }
