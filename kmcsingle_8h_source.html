111 void KMCSingle::LoadGraph() {
112 
113  Database db;
114  db.Open( _filename );
115  cout << " Loading graph from " << _filename << endl;
116  Statement *stmt = db.Prepare("SELECT _id, name FROM segments;");
117 
118  while (stmt->Step() != SQLITE_DONE) {
119  int id = stmt->Column<int>(0);
120  string name = stmt->Column<string>(1);
121  node_t *n =new node_t(id);
122  _nodes.push_back(n);
123  _nodes_lookup[id] = _nodes.back();
124  if (wildcmp(_injection_name.c_str(), name.c_str())) {
125  _injection.push_back(n);
126  _injection_lookup[id] = _injection.back();
127  }
128  }
129  //delete stmt;
130  cout << " -Nodes: " << _nodes.size() << endl;
131  cout << "seed:" << _seed << endl;
132  if(_seed > _nodes.size()){ throw invalid_argument ("Seed is bigger than number of nodes. Please specify a smaller seed in your input file."); }
133  cout << " -Injection Points: " << _injection.size() << endl;
134 
135  delete stmt;
136 
137  int links = 0;
138  stmt = db.Prepare("SELECT seg1, seg2, rate12e, rate21e, drX, drY, drZ FROM pairs;"); // electron rates, check (think about) this
139  while (stmt->Step() != SQLITE_DONE) {
140  node_t *n1 = _nodes_lookup[stmt->Column<int>(0)];
141  node_t *n2 = _nodes_lookup[stmt->Column<int>(1)];
142  double rate12 = stmt->Column<double>(2);
143  double rate21 = stmt->Column<double>(3);
144  vec r = vec(stmt->Column<double>(4), stmt->Column<double>(5), stmt->Column<double>(6));
145  n1->AddEvent(new link_t(n2, rate12, r));
146  n2->AddEvent(new link_t(n1, rate21, -r));
147  links += 2;
148 
149  if ( votca::tools::globals::verbose ) {
150  cout << "rate12=" << rate12 << endl;
151  cout << "rate21=" << rate21 << endl;
152  cout << "r=" << r << endl;
153  }
154  }
155  delete stmt;
156  cout << " -Links: " << links << endl;
157 
158 }

160 void KMCSingle::RunKMC(void)
161 {
162 
163  double t = 0;
164 
165  srand(_seed);
166  votca::tools::Random::init(rand(), rand(), rand(), rand());
167 
168  // cout << " seed:size:site " << _seed << ":" << _injection.size() << ":" << Random::rand_uniform_int(_injection.size()) << endl;
169  current=_injection[Random::rand_uniform_int(_injection.size())];
170  cout <<" Starting simulation at node: "<<current->_id-1<<endl;
171  if (current->Rate() == 0) {
172  int id = current->_id;
173  throw std::runtime_error("Injected on an unconnected site, id = "+boost::lexical_cast<string>(id));
174  }
175  double next_output = _dt;
176  int i=0;
177  while(t<_runtime) {
178  t+=current->WaitingTime();
179  current->onExecute(); // this line causes a Segmentation fault
180  if(t > next_output) {
181  next_output = t + _dt;
182  cout << t << ": " << r << endl;
183  }
184  }
185  _runtime = t;
186  WriteOcc();
187  cout << std::scientific << "\nKMC run finished\nAverage velocity (m/s): " << r/t*1e-9 << endl;
188 }

190 void KMCSingle::WriteOcc()
191 {
192  Database db;
193  cout << "Opening for writing " << _filename << endl;
194  db.Open(_filename);
195  db.Exec("BEGIN;");
196  Statement *stmt = db.Prepare("UPDATE segments SET occPe = ? WHERE _id = ?;"); // electron occ. prob., check (think about) this
197  for(int i=0; i<_nodes.size(); ++i) {
198  stmt->Reset();
199  stmt->Bind(1, _nodes[i]->_occ/_runtime);
200  stmt->Bind(2, _nodes[i]->_id);
201  stmt->Step();
202  }
203  db.Exec("END;");
204  delete stmt;
205 }
