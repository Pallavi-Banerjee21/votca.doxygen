30 bool DLPOLYTrajectoryReader::Open(const string &file)
31 {
32  boost::filesystem::path filepath(file.c_str());
33  string filename;
34 
35  if ( boost::filesystem::basename(filepath).size() == 0 ) {
36  if (filepath.parent_path().string().size() == 0) {
37  filename="HISTORY";
38  } else {
39  filename=filepath.parent_path().string() + "/HISTORY";

42  filename=file;

44 
45  _fl.open(filename.c_str());

64 bool DLPOLYTrajectoryReader::NextFrame(Topology &top)
65 {
66  static int mavecs=0;
67  static int mpbct=0;
68  static int matoms=0;
69  static bool hasVs=false;
70  static bool hasFs=false;
71 
72  string line;
73 
74  BoundaryCondition::eBoxtype pbc_type=BoundaryCondition::typeAuto;
75 
76  if (_first_frame) {
77  getline(_fl, line); //title
78  cout << "Read from HISTORY: '" << line << "' - header" << endl;
79 
80  getline(_fl, line); // 2nd header line
81  cout << "Read from HISTORY: '" << line << "' - directives line" << endl;
82 
83  Tokenizer tok(line, " \t");
84  vector<int> fields;
85  tok.ConvertToVector<int>(fields);
86 
87  mavecs = boost::lexical_cast<int>(fields[0]);
88  mpbct = boost::lexical_cast<int>(fields[1]);
89  matoms = boost::lexical_cast<int>(fields[2]);
90 
91  //if(mavecs > 0 ) hasVs=true;
92  //if(mavecs > 1 ) hasFs=true;
93  hasVs=(mavecs > 0);
94  hasFs=(mavecs > 1);
95 
96  if(hasVs != top.HasVel() || hasFs != top.HasForce()) {
97  cout << "Warning: TrajKey (# of atom vectors) in HISTORY (trajectory) header & CONFIG (initial frame) differ" << endl;
98  top.SetHasVel(hasVs);
99  top.SetHasForce(hasFs);
100  }
101  cout << "Read from HISTORY: traj_key - " << mavecs << ", hasV - " << top.HasVel() << ", hasF - " << top.HasForce() << endl;
102 
103  //if(fields[2] != top.BeadCount())
104  if(matoms != top.BeadCount())
105  throw std::runtime_error("Number of atoms/beads in HISTORY (trajectory) header & CONFIG (initial frame) differ");
106 
107  if(mpbct == 0) {
108  pbc_type=BoundaryCondition::typeOpen;
109  }
110  else if(mpbct == 1 || mpbct == 2 ) {
111  pbc_type=BoundaryCondition::typeOrthorhombic;
112  }
113  else if(mpbct == 3) {
114  pbc_type=BoundaryCondition::typeTriclinic;
115  }
116  cout << "Read from HISTORY: pbc_type - '" << pbc_type << "'" << endl;
117 
118  if(pbc_type != top.getBoxType())
119  //throw std::runtime_error("PBC type in HISTORY (trajectory) header & CONFIG (initial frame) differ");
120  cout << "Warning: PBC type in HISTORY (trajectory) header & CONFIG (initial frame) differ" << endl;
121  }
122 
123  //read normal frame
124  getline(_fl, line); // timestep line
125  //cout << "Read from HISTORY: '" << line << "'" << endl;
126 
127  if(!_fl.eof()) {
128  double dtime,stime,scale;
129  int nstep;
130  int natoms;
131  int navecs;
132  int npbct;
133 
134  //scale = 0.1; // AB: factor to convert Angstroem to nm
135  scale = 1.0; // AB: factor to convert Angstroem to nm - not needed?
136 
137  {
138  Tokenizer tok(line, " \t");
139  vector<string> fields;
140  tok.ToVector(fields);
141 
142  nstep = boost::lexical_cast<int>(fields[1]);
143  natoms = boost::lexical_cast<int>(fields[2]);
144  navecs = boost::lexical_cast<int>(fields[3]);
145  npbct = boost::lexical_cast<int>(fields[4]);
146  dtime = boost::lexical_cast<double>(fields[5]);
147  stime = boost::lexical_cast<double>(fields[fields.size()-1]);
148 
149  //cout << "Read from HISTORY: natoms = " << natoms << ", levcfg = " << fields[3];
150  //cout << ", dt = " << fields[5] << ", time = " << stime << endl;
151 
152  // AB-TODO: would be nice to store navecs (levcfg) & npbct (imcon, PBC switch)
153 
154  if(natoms != top.BeadCount())
155  throw std::runtime_error("number of atoms/beads in HISTORY (trajectory) frame & FIELD (topology) differ");
156  if(navecs != mavecs)
157  throw std::runtime_error("TrajKey (# of atom vectors) in HISTORY (trajectory) header & frame differ");
158  if(natoms != matoms)
159  throw std::runtime_error("Number of atoms/beads in HISTORY (trajectory) header & frame differ");
160  if(npbct != mpbct)
161  throw std::runtime_error("PBC type in HISTORY (trajectory) header & frame differ");
162 
163  top.setTime(nstep*dtime);
164  //top.setTime(boost::lexical_cast<double>(fields[5]));
165 
166  //if(stime != top.getTime() )
167  //cout << "Check: nstep = " << nstep << ", dt = " << dtime << ", time = " << top.getTime() << " (correct?)" << endl;
168  }
169 
170  vec box_vectors[3];
171  for (int i=0;i<3;i++){ // read 3 box lines
172  getline(_fl, line);
173  //cout << "Read from HISTORY: '" << line << "' - box vector # " << i+1 << endl;
174 
175  if(_fl.eof())
176  throw std::runtime_error("unexpected EOF in HISTOPY (trajectory), when reading box vector"+
177  boost::lexical_cast<string>(i));
178 
179  Tokenizer tok(line, " \t");
180  vector<double> fields;
181  tok.ConvertToVector<double>(fields);
182  box_vectors[i]=vec(fields[0]*scale,fields[1]*scale,fields[2]*scale);
183  }
184  matrix box(box_vectors[0],box_vectors[1],box_vectors[2]);
185 
186  if(npbct == 0) {
187  pbc_type=BoundaryCondition::typeOpen;
188  }
189  else if(npbct == 1 || npbct == 2 ) {
190  pbc_type=BoundaryCondition::typeOrthorhombic;
191  }
192  else if(npbct == 3) {
193  pbc_type=BoundaryCondition::typeTriclinic;
194  }
195 
196  top.setBox(box,pbc_type);
197 
198  for (int i=0;i<natoms;i++){
199 
200  {
201  getline(_fl, line); //atom header line
202  //cout << "Read from HISTORY: '" << line << endl;
203 
204  if(_fl.eof())
205  throw std::runtime_error("unexpected EOF in HISTOPY (trajectory), when reading atom/bead nr" +
206  boost::lexical_cast<string>(i+1));
207 
208  vector<string> fields;
209  Tokenizer tok(line, " \t");
210  tok.ToVector(fields);
211  int id=boost::lexical_cast<double>(fields[1]);
212  if (i+1 != id )
213  throw std::runtime_error("unexpected atom/bead index in HISTOPY (trajectory), expected" +
214  boost::lexical_cast<string>(i+1) + "got" + boost::lexical_cast<string>(id));
215  }
216 
217  Bead *b = top.getBead(i);
218  vec atom_vec[3];
219  for (int j=0;j<min(navecs,2)+1;j++){
220  getline(_fl, line); //read atom positions
221  if(_fl.eof())
222  throw std::runtime_error("unexpected EOF in HISTOPY (trajectory) when reading atom/bead vector" +
223  boost::lexical_cast<string>(j) + " of atom " + boost::lexical_cast<string>(i+1) );
224  vector<double> fields;
225  Tokenizer tok(line, " \t");
226  tok.ConvertToVector<double>(fields);
227  if(j<3) {
228  // AB: convert Angstroem to nm; j=1 => x,y,z; j=2 => vx,vy,vz
229  atom_vec[j]=vec(fields[0]*scale,fields[1]*scale,fields[2]*scale);
230  }
231  else { // AB: forces do not need to be converted (assuming kJ/mole?)
232  atom_vec[j]=vec(fields[0],fields[1],fields[2]);
233  }
234  }
235  b->setPos(atom_vec[0]);
236  if (navecs > 0)
237  b->setVel(atom_vec[1]);
238  if (navecs > 1)
239  b->setF(atom_vec[2]);
240  }
241  }
242  return !_fl.eof();
243 }
